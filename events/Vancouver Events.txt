vim.6 = {
	#Thousands of American gold miners arrive
	type = country_event
	placement = scope:vim_gold_mine_state
	event_image = {
		video = "gfx/event_pictures/unspecific_trains.bk2"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/trains"
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	title = vim.2.t
	desc = vim.2.dsc
	flavor = vim.2.f
	duration = 3
	trigger = {
		exists = c:VCI
		THIS = c:VCI
		game_date < 1858.6.1
		NOT = {
			has_law = law_type:law_closed_borders
		}
		any_scope_state = {
			has_building = building_gold_mine
		}
		NOT = {
			has_variable = vim_event_6
		}
	}
	immediate = {
		set_variable = vim_event_6
		random_scope_state = {
			limit = {
				owner = c:VCI
				has_building = building_gold_mine
			}
			save_scope_as = vim_gold_mine_state
		}
		scope:vim_gold_mine_state = {
			create_pop = {
				pop_type = labors
				culture = cu:han
				size = 300
			}
		}
	}
	option = {
		name = vim.2.a
		add_treasury = 50000
		add_loyalists = {
			culture = cu:han
			value = 0.05
		}
	}
	option = {
		name = vim.2.a
		default_option = yes
		add_treasury = -50000
		if = {
			limit = {
				exists = c:USA
			}
			change_relations = {
				country = c:USA
				value = -10
			}
		}
		if = {
			limit = {
				exists = c:CSA
			}
			change_relations = {
				country = c:CSA
				value = -10
			}
		}
		scope:vim_gold_mine_state = {
			add_modifier = {
				name = vim_immigrants
				year = 10
			}
			create_pop = {
				pop_type = labors
				culture = cu:han
				size = 1000
			}
			create_pop = {
				pop_type = labors
				culture = cu:dixie
				size = 3000
			}
		}
	}
}
